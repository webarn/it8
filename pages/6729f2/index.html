    # 代码题

    实现一个函数 add(1,2);add(1)(2);add(1,2)(3); 都可以返回参数相加的值

    function add() {
  let args = Array.prototype.slice.call(arguments);
  let inner = function () {
    args = args.concat(...arguments);
    return inner;
  };
  inner.toString = function () {
    return args.reduce((pre, cur) => {
      return pre + cur;
    }, 0);
  };
  return inner;
}

let r = add(1, 2, 3)(4);
console.log(typeof r); // function: 这里返回的是一个方法
console.log(Number(r)); // 10
console.log(r + 1); // 11

    function curr(protocol, host, params) {
  let args = [...arguments];
  let inner = function () {
    args = args.concat(...arguments);
    return `${args[0]}://${args[1]}?search=${args[2]}`;
  };
  return inner;
}

const protocol = curr('https');
const r = protocol('www.it8.tech', 'hello');

    # es6 实现

    function curr(func, ...args) {
  return function (...subArgs) {
    const allArgs = args.concat(subArgs);
    if (allArgs.length >= func.length) {
      return func(...allArgs);
    } else {
      return curr(func, ...allArgs);
    }
  };
}

function add(a, b, c) {
  return a + b + c;
}

let r = curr(add);

const com = r(5);
console.log(com(2)(3));
