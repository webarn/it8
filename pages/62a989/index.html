    实现some函数

    # 使用 es5 实现 some

    var arr = [1, 2, 3, 4, 5, 6];
Array.prototype._some = function (fn) {
  if (Object.prototype.toString.call(this) !== '[object Array]') {
    throw Error('array');
  }
  for (var i = 0; i < this.length; i++) {
    let r = fn.call(this, this[i]);
    if (r) return true;
  }
  return false;
};

let r = arr._some(function (item) {
  return item > 4;
});

console.log(r);
