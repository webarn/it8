数组对象求平均值

对一群学生的所有数值属性（整形或浮点）求均值。要求： 1. 属性 dict/object 嵌套层级 N 层（不限于例子 3 层）。 2. 结果数据四舍五入，保留 2 位小数。

// 判断是否为数值
function isNumber(num) {
  const type = typeof num;

  if (type === 'string') {
    return !Number.isNaN(+num);
  } else if (type === 'number') {
    return true;
  } else {
    return false;
  }
}

function testListWithNestedDicts(data = []) {
  let res = {};
  let temp = [];

  data.forEach((items, index) => {
    for (const [key, val] of Object.entries(items)) {
      //
      if (isNumber(val)) {
        if (!res[key]) res[key] = 0;
        // 计算平均值： (已记录平均值 * 已记录长度 = 原数值)， (原始值 + 新增值) / 新长度
        let sub = (res[key] * index + val) / (index + 1);
        res[key] = sub.toFixed(2);
      } else {
        // 这里利用了复杂数据的引用,利用递归
        temp.push(val);
        // console.log(temp);
        res[key] = testListWithNestedDicts(temp);
        console.log(res[key]);
      }
    }
  });
  return res;
}

const res = testListWithNestedDicts(arr);
