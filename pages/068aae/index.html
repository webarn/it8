    十进制转换二进制

    被除数不断除以 2 再取余,得到的余数压入栈中,被除数的结果再进入下一次循环计算

    # 方法一

    function decode2bin(num) {
  if (isNaN(num)) {
    throw new Error('请输入一个数字');
  }

  var arr = [];
  while (num != 0) {
    var r = num % 2;
    console.log('r: ', r);
    arr[arr.length] = r;
    num = parseInt(num / 2);
  }
  //将 arr 反转
  arr.reverse();
  //使用 "" 拼接 arr 数组
  var result = arr.join('');
  return result;
}

const r = decode2bin(3);
console.log(r); // 11

    # 方法二

    function dec2bin(num) {
  let arr = [];
  while (num > 0) {
    arr.push(num % 2);
    num = Math.floor(num / 2);
  }

  const result = arr.reverse().join('');

  return result;
}

dec2bin(100);
