    观察者、发布订阅模式

    # 观察者模式

    // 被观察对象
class Baby {
  constructor(name) {
    this.name = name;
    this.state = '开心';
    this.Observers = [];
  }

  attach(who) {
    this.Observers.push(who);
  }

  setState(state) {
    this.state = state;
    this.Observers.forEach((who) => {
      who.update(this);
    });
  }
}

// 观察者对象
class Observer {
  constructor(name) {
    this.name = name;
  }

  update(sub) {
    console.log(`${this.name}观察到${sub.name}的状态为${sub.state}`);
  }
}

const baby = new Baby('小宝宝');

const father = new Observer('爸爸');
const mother = new Observer('妈妈');

baby.attach(father);
baby.attach(mother);

baby.setState('灰常开心');

    # 发布订阅模式

    const e = {
  _callback: [],
  on(cb) {
    this._callback.push(cb);
  },
  emit(val) {
    this._callback.forEach((cb) => cb(val));
  },
};

e.on(function (val) {
  console.log('1' + val);
});
e.on(function (val) {
  console.log('2' + val);
});
e.on(function (val) {
  console.log('3' + val);
});

e.emit('发布');
