    # 分析

    1. 需要返回一个函数
    2. 执行次数是否大于 0
    3. 每隔 ? 秒 需要使用定时器 重复调用 并执行方法
    4. 判断停止条件
    5. 手动取消

    # 实现

    function repeat(func, times, wait) {
  if (times <= 0) return;
  let t = 0;
  let timer = null;
  const inner = function (...args) {
    timer = setTimeout(() => {
      if (t++ === times) {
        clearTimeout(timer);
        return;
      }
      func.apply(this, args);
      inner.apply(this, args);
    }, wait);
  };

  inner.clear = function () {
    console.log(timer);
    clearTimeout(timer);
    times = 0;
  };

  return inner;
}

const repeatFunc = repeat(alert, 4, 3000);
repeatFunc('hellworld');
