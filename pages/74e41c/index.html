// map方式
function toTree(arr) {
  const result = [],
    map = {};
  for (const item of arr) {
    const { id, parentId } = item;
    map[id] = {
      ...item,
      children: map[id] ? map[id].children : [],
    };
    const newItem = map[id];

    if (parentId) {
      if (!map[parentId]) {
        map[parentId] = { children: [] };
      }
      map[parentId].children.push(newItem);
    } else {
      result.push(item);
    }
  }
  return result;
}

function flat(data) {
  data.reduce((pre, next) => {
