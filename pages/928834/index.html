    Promise方法的实现

    # 分析

    步骤分析

    1. promise 有三个状态：pending，fulfilled，or rejected;

    2. new promise 时， 需要传递一个 executor()执行器，执行器立即执行;

    3. executor 接受两个参数，分别是 resolve 和 reject;

    4. promise 的默认状态是 pending;

    5. promise 只能从 pending 到 rejected, 或者从 pending 到 fulfilled，状态一旦确认，就不会再改变;

    6. promise 需要有一个值来接受结果 result;

    7. promise 必须有一个 then 方法，then 接收两个参数，分别是 promise 成功的回调 onFulfilled, 和 promise 失败的回调 onRejected;

    8. 如果调用 then 时，promise 已经成功，则执行 onFulfilled, 参数是 promise 的 this.state;

    9. 如果调用 then 时，promise 已经失败，则执行 onRejected, 参数是 promise 的 this.state;

    10. 创建实例的时候是否有报错信息, 如果有需要抛出;

    11. 如果 then 方法传入的不是一个方法,则需要设置为一个默认方法;

    12. 现在为止 promise 还是同步执行, 并不能异步操作;

    13. 创建待执行数组 onResolvedCallbacks onRejectedCallbacks 先入先出的顺序;

    14. 如果 then 的时候状态为待执行, 则把方法放在待执行数组内;

    15. 在 resolve reject 方法内遍历待执行数组,并执行;

    16. 如果 第 5 行代码 resolve 后还有代码, 则后面的代码会等到 resolve 执行完毕后才会执行:

    # Promise 代码实现

    class Promise_diy {
  // 三种状态
  static PENDING = 'pending'; // 待执行
  static FULFILLED = 'fulfilled'; // 成功
  static REJECTED = 'rejected'; // 失败
  constructor(executor) {
    this.state = Promise_diy.PENDING; // promise 的默认状态是 pending；
    this.result = null; // promise 需要有一个值来接受结果
    this.onResolvedCallbacks = []; //创建待执行数组 onResolvedCallbacks
    this.onRejectedCallbacks = []; // 创建待执行数组 onRejectedCallbacks

    // 创建实例的时候是否有报错信息, 如果有需要抛出;
    try {
      // 注意this,需要绑定
      executor(this.resolve.bind(this), this.reject.bind(this)); // 传递一个 executor()执行器，执行器立即执行, 接受两个参数，分别是 `resolve` 和 `reject`;
    } catch (error) {
      this.reject(error); // 这里不需要绑定this, 因为是在实例中执行的
    }
  }

  resolve(result) {
    // 加入setTimeout保证后面的代码输出: 解决 16 的问题
    setTimeout(() => {
      // promise 只能从 pending 到 rejected, 或者从 pending 到 fulfilled，状态一旦确认，就不会再改变;
      if (this.state === Promise_diy.PENDING) {
        this.state = Promise_diy.FULFILLED;
        this.result = result;

        // 遍历待执行数组,并并执行;
        this.onResolvedCallbacks.forEach((callback) => {
          callback(result);
        });
      }
    });
  }

  reject(result) {
    // 加入setTimeout保证后面的代码输出: 解决 16 的问题
    setTimeout(() => {
      // promise 只能从 pending 到 rejected, 或者从 pending 到 fulfilled，状态一旦确认，就不会再改变;
      if (this.state === Promise_diy.PENDING) {
        this.state = Promise_diy.REJECTED;
        this.result = result;

        // 遍历待执行数组,并并执行;
        this.onRejectedCallbacks.forEach((callback) => {
          callback(result);
        });
      }
    });
  }

  // promise 必须有一个 then 方法，then 接收两个参数，分别是 promise 成功的回调 onFulfilled, 和 promise 失败的回调 onRejected;
  then(onFulfilled, onRejected) {
    // 实现 then 方法后链式调用 then, 在 then 方法里返回一个新的 Promise 实例
    return new Promise_diy((resolve, reject) => {
      // 如果 then 方法传入的不是一个方法,则需要设置为一个默认方法;
      onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : () => {};
      onRejected = typeof onRejected === 'function' ? onRejected : () => {};

      // 如果状态为待执行,则放进一个待执行数组内
      if (this.state === Promise_diy.PENDING) {
        this.onResolvedCallbacks.push(onFulfilled);
        this.onRejectedCallbacks.push(onRejected);
      }

      // 如果调用 then 时，promise 已经成功，则执行 onFulfilled，参数是 promise 的 state
      if (this.state === Promise_diy.FULFILLED) {
        onFulfilled(this.state);
      }

      // 如果调用 then 时，promise 已经失败，那么执行 onRejected, 参数是 promise 的 state
      if (this.state === Promise_diy.REJECTED) {
        onRejected(this.state);
      }
    });
  }
}

const p = new Promise_diy((resolve, reject) => {
  console.log(1);
  setTimeout(() => {
    console.log(5);
    resolve(2);
    console.log(6);
  }, 2000);
  console.log(3);
});

p.then(

    console.log('res: ', res);
  },
  (err) => {
    console.log('err: ', err);
  },
).then((res) => {
  console.log(res);
});

    # Promise.all 的实现

    # 特点:

    1. 接收一个 Promise 实例的数组或具有 Iterator 接口的对象
    2. 如果元素不是 Promise 对象，则使用 Promise.resolve 转成 Promise 对象
    3. 如果全部成功，状态变为 resolved，返回值将组成一个数组传给回调
    4. 只要有一个失败，状态就变为 rejected，返回值将直接传递给回调 all() 的返回值也是新的 Promise 对象
    const h1 = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('success-h1');
  }, 1000);
});

const h2 = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('success-h2');
  }, 2000);
});

const h3 = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('success-h3');
  }, 3000);
});

function Promise_all(httpArr) {
  return new Promise((resolve, reject) => {
    if (!isArray(promises)) {
      return reject(new TypeError('arguments must be an array'));
    }

    let result = []; // 存放结果
    let j = 0; // 判断结束条件

    for (let i = 0; i < httpArr.length; i++) {
      // 使用 Promise.resolve 把不是Promise的转换为Promise
      Promise.resolve(httpArr[i]).then(
        (res) => {
          j++;
          result[i] = res;
          if (j === httpArr.length) {
            resolve(result);
          }
        },
        (err) => {
          reject(err);
        },
      );
    }
  });
}

  })

  });
